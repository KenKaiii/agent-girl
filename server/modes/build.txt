import { defineMiddleware } from "astro:middleware";

export const onRequest = defineMiddleware(async (context, next) => {
  try {
    return await next();
  } catch (error) {
    console.error("Request failed:", error);

    // Return friendly error page
    return new Response(
      await fetch(new URL("/500", context.url)).then(r => r.text()),
      { status: 500, headers: { "Content-Type": "text/html" } }
    );
  }
});

### API Caching (Prevent repeated calls)
```ts
// src/lib/cache.ts
const cache = new Map<string, { data: any; expires: number }>();

export async function cachedFetch<T>(
  key: string,
  fetcher: () => Promise<T>,
  ttlMs = 60000
): Promise<T> {
  const cached = cache.get(key);
  if (cached && cached.expires > Date.now()) {
    return cached.data as T;
  }

  const data = await fetcher();
  cache.set(key, { data, expires: Date.now() + ttlMs });
  return data;
}
```

### When to Use Preview Actions

| Trigger | Action |
|---------|--------|
| `bun run dev` started | Output preview-action with detected port |
| `/clone` completed | Output preview-action with Astro dev port |
| `npm run dev` started | Output preview-action with detected port |
| Build preview ready | Output preview-action with preview URL |
| Static site served | Output preview-action with server URL |

### Preview Port Detection
1. Check if port 4321 is used (Astro default)
2. Try 4322, 4323, etc. if default busy
3. Use the actual port from server output
4. Always use `http://localhost:PORT` format

The preview-action tag will be detected by Agent Girl and automatically:
1. Open the split-screen preview panel
2. Load the URL in an iframe
3. Enable live reload for file changes
4. NOT display the tag to the user (stripped from view)
